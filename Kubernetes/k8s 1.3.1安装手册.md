# 安装最新的K8S集群，底层使用Docker作为容器运行时



### 1. 虚拟机节点的规划和操作系统的安装

虚拟机使用vagrant来批量创建虚拟机，本次实验使用vagrant创建三台虚拟机，虚拟机的规划为：

|    主机名    |    IP地址    |
| :----------: | :----------: |
| k8s-master01 | 192.168.8.8  |
| k8s-worker01 | 192.168.8.9  |
| K8s-worker02 | 192.168.8.10 |

Vagrant创建脚本如下：

```Ruby
vm_list = [
  { # hash map
    "name" => "k8s-master01",
    "cpu" => "2",
    "mem" => "2048",
    "ip_addr" => "192.168.8.8"
  },
  {
    "name" => "k8s-worker01",
    "cpu" => "2",
    "mem" => "2048",
    "ip_addr" => "192.168.8.9"
  },
  {
    "name" => "k8s-worker02",
    "cpu" => "2",
    "mem" => "2048",
    "ip_addr" => "192.168.8.10"
  }
]

Vagrant.configure(2) do |config|

   config.vm.box = "/Users/guangyongluo/Downloads/CentOS-7-x86_64-Vagrant-2004_01.VirtualBox.box"

   vm_list.each do |item|
        config.vm.define item["name"] do |node|

            node.vm.provider "virtualbox" do |vbox|
              vbox.name = item["name"];  # 虚拟机名称
              vbox.memory = item["mem"]; # 内存
              vbox.cpus = item["cpu"];   # CPU
            end
            # 设置hostanme
            node.vm.hostname = item["name"]
            # 设置IP
            node.vm.network "private_network", ip: item["ip_addr"]
            # 设置root密码
            config.vm.provision "shell", inline: <<-SHELL
    	      echo 'root:root@123' | chpasswd
            SHELL
        end
    end
end

```



### 2. 对CentOS7系统进行内核升级，修改相关系统配置

```shell
# 导入公钥
rpm --import https://www.elrepo.org/RPM-GPG-KEY-elrepo.org  

# 安装elrepo YUM源仓库
yum install https://www.elrepo.org/elrepo-release-7.0-4.el7.elrepo.noarch.rpm
yum install https://www.elrepo.org/elrepo-release-8.el8.elrepo.noarch.rpm



# 安装kernel-ml版本，ml为长期稳定版本，lt为长期维护版本
yum --enablerepo="elrepo-kernel" -y install kernel-lt.x86_64


# 设置grub2默认引导为0
grub2-set-default 0

# 重新生成grub2引导文件
grub2-mkconfig -o /boot/grub2/grub.cfg

# 系统重启
reboot

# 关闭防火墙
systemctl stop firewalld
systemctl disable firewalld

# 关闭selinux
sed -i 's/enforcing/disabled/' /etc/selinux/config  # 永久
setenforce 0  # 临时

# 关闭swap
swapoff -a  # 临时
sed -ri 's/.*swap.*/#&/' /etc/fstab    # 永久

# 关闭完swap后，一定要重启一下虚拟机！！！
# 根据规划设置主机名
# hostnamectl set-hostname <hostname>

# 在个节点添加hosts映射
cat >> /etc/hosts << EOF
192.168.8.8 k8s-master01
192.168.8.9 k8s-worker01
192.168.8.10 k8s-worker02
EOF

# 将桥接的IPv4流量传递到iptables的链
cat > /etc/sysctl.d/k8s.conf << EOF
net.bridge.bridge-nf-call-ip6tables = 1
net.bridge.bridge-nf-call-iptables = 1
net.ipv4.ip_forward = 1
vm.swappiness = 0
EOF

# 加载br_netfilter模块
modprobe br_netfilter

sysctl --system  # 生效

# 安装ip_vs
yum -y install ipset ipvsadm

# 加载ip_vs模块
cat > /etc/sysconfig/modules/ipvs.modules <<EOF
#!/bin/bash
modprobe -- ip_vs
modprobe -- ip_vs_rr
modprobe -- ip_vs_wrr
modprobe -- ip_vs_sh
modprobe -- nf_conntrack
EOF
# 授权、运行、检查是否加载
chmod 755 /etc/sysconfig/modules/ipvs.modules && bash /etc/sysconfig/modules/ipvs.modules && lsmod | grep -e ip_vs -e nf_conntrack


# 时间同步
yum install ntpdate -y
ntpdate time.windows.com

# centos8时间同步设置
vi /etc/chrony.conf

# 注释掉：
# pool 2.centos.pool.ntp.org iburst
# 加入新的的时间服务器（可添加多个时间服务器）：
pool ntp.aliyun.com iburst

# 重启 chronyd 服务
systemctl restart chronyd.service

# 设置开机自启动
systemctl enable chronyd.service
```

### 3. 安装docker-ce和cri-docker

```shell
# 配置aliyun的安装源
yum-config-manager --add-repo http://mirrors.aliyun.com/docker-ce/linux/centos/docker-ce.repo

# 创建或者更新缓存
yum makecache

# 安装docker-ce
yum install -y docker-ce

# 添加docker配置信息
cat > /etc/docker/daemon.json <<EOF
{
    "exec-opts":["native.cgroupdriver=systemd"]
}
EOF

# 开启docker开机自启动
systemctl enable --now docker

# 下载cri-dockerd
wget https://github.com/Mirantis/cri-dockerd/releases/download/v0.3.14/cri-dockerd-0.3.14-3.el7.x86_64.rpm

# 安装cri-dockerd
rpm -ivh cri-dockerd-0.3.14-3.el7.x86_64.rpm

# 指定cri-dockerd服务的基础镜像
# 查看cri-dockerd服务的配置文件
systemctl status cri-docker
# 修改配置文件，添加使用pause:3.9作为Pod基础容器服务
# ExecStart=/usr/bin/cri-dockerd --pod-infra-container-image=registry.k8s.io/pause:3.9 --container-runtime-endpoint fd://

# 启动cri-dockerd
systemctl enable --now cri-docker

```

### 4. 安装kubeadm, kubelet和kubectl

```shell
# kubernetes yum源的准备，使用aliyun
cat <<EOF | tee /etc/yum.repos.d/kubernetes.repo
[kubernetes]
name=Kubernetes
baseurl=https://mirrors.aliyun.com/kubernetes-new/core/stable/v1.30/rpm/
enabled=1
gpgcheck=1
gpgkey=https://mirrors.aliyun.com/kubernetes-new/core/stable/v1.30/rpm/repodata/repomd.xml.key
EOF

# 安装kubeadm、kubelet和kubectl
yum -y install kubeadm kubelet kubectl

# 为了实现docker使用的cgroupdriver与kubelet使用的cgroup一致，修改kubelet配置文件/etc/sysconfig/kubelet
KUBELET_EXTRA_ARGS="--cgroup-driver=systemd"

# 初始化kubelet服务
systemctl enable kubelet

# 拉取kubeadm所需的所有镜像
docker pull registry.cn-hangzhou.aliyuncs.com/google_containers/kube-apiserver:v1.30.2
docker pull registry.cn-hangzhou.aliyuncs.com/google_containers/kube-controller-manager:v1.30.2
docker pull registry.cn-hangzhou.aliyuncs.com/google_containers/kube-scheduler:v1.30.2
docker pull registry.cn-hangzhou.aliyuncs.com/google_containers/kube-proxy:v1.30.2
docker pull registry.cn-hangzhou.aliyuncs.com/google_containers/coredns:v1.11.1
docker pull registry.cn-hangzhou.aliyuncs.com/google_containers/pause:3.9
docker pull registry.cn-hangzhou.aliyuncs.com/google_containers/etcd:3.5.12-0

# 重命名所有kubeadm的所需镜像
docker tag registry.cn-hangzhou.aliyuncs.com/google_containers/kube-apiserver:v1.30.2 registry.k8s.io/kube-apiserver:v1.30.2
docker tag registry.cn-hangzhou.aliyuncs.com/google_containers/kube-controller-manager:v1.30.2 registry.k8s.io/kube-controller-manager:v1.30.2
docker tag registry.cn-hangzhou.aliyuncs.com/google_containers/kube-scheduler:v1.30.2 registry.k8s.io/kube-scheduler:v1.30.2
docker tag registry.cn-hangzhou.aliyuncs.com/google_containers/kube-proxy:v1.30.2 registry.k8s.io/kube-proxy:v1.30.2
docker tag registry.cn-hangzhou.aliyuncs.com/google_containers/coredns:v1.11.1 registry.k8s.io/coredns/coredns:v1.11.1
docker tag registry.cn-hangzhou.aliyuncs.com/google_containers/pause:3.9 registry.k8s.io/pause:3.9
docker tag registry.cn-hangzhou.aliyuncs.com/google_containers/etcd:3.5.12-0 registry.k8s.io/etcd:3.5.12-0

# 初始化K8S集群
kubeadm init --kubernetes-version=v1.30.2 --pod-network-cidr=10.10.8.0/16 --apiserver-advertise-address=192.168.8.8 --cri-socket unix:///var/run/cri-dockerd.sock

# 加入worker节点
kubeadm join 192.168.8.8:6443 --token ys6imd.nxikj59p9iq1tbnf --discovery-token-ca-cert-hash sha256:f7018966f8e3f649ab6eda8b4de038e7da7c2929c2c2a89fa9c76ac430dfd01d --cri-socket unix:///var/run/cri-dockerd.sock

# 查看所有node的状态
[root@k8s-master01 ~]# kubectl get nodes
NAME           STATUS     ROLES           AGE    VERSION
k8s-master01   NotReady   control-plane   8m9s   v1.30.2
k8s-worker01   NotReady   <none>          40s    v1.30.2
k8s-worker02   NotReady   <none>          12s    v1.30.2
```
